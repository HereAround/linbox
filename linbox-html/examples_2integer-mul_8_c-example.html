// ---------------------------------------------
#include <iostream>
#include <fstream>
// ---------------------------------------------
#include <linbox/linbox-config.h>
// Use of Gmp based LinBox integers
#include <linbox/integer.h>
using namespace LinBox;
using namespace std;
// ---------------------------------------------
int main()
{
integer a,b;
cout << "1st integer > ";
cin >> a;
cout << "2nd integer > ";
cin >> b;
cout << "The product " << a*b << "\n";
return 0;
}
